37 double PerturbativeInteraction::getC6(StateTwo state, double deltaN) {
38  double C6 = 0;
39 
40  for (int n0 = state.n[0] - deltaN; n0 <= state.n[0] + deltaN; ++n0) {
41 
42  for (int n1 = state.n[1] - deltaN; n1 <= state.n[1] + deltaN; ++n1) {
43 
44  std::vector<int> array_l0;
45  if (state.l[0] > 0) {
46  array_l0.push_back(state.l[0] - 1);
47  }
48  if (state.l[0] < n0 - 1) {
49  array_l0.push_back(state.l[0] + 1);
50  }
51  for (int l0 : array_l0) {
52 
53  std::vector<int> array_l1;
54  if (state.l[1] > 0) {
55  array_l1.push_back(state.l[1] - 1);
56  }
57  if (state.l[1] < n1 - 1) {
58  array_l1.push_back(state.l[1] + 1);
59  }
60  for (int l1 : array_l1) {
61 
62  std::set<float> array_j0;
63  if (std::abs(std::abs(l0 - state.s[0]) - state.j[0]) < 2) {
64  array_j0.insert(std::abs(l0 - state.s[0]));
65  }
66  if (std::abs(l0 + state.s[0] - state.j[0]) < 2) {
67  array_j0.insert(l0 + state.s[0]);
68  }
69  for (float j0 : array_j0) {
70 
71  std::set<float> array_j1;
72  if (std::abs(std::abs(l1 - state.s[1]) - state.j[1]) < 2) {
73  array_j1.insert(std::abs(l1 - state.s[1]));
74  }
75  if (std::abs(l1 + state.s[1] - state.j[1]) < 2) {
76  array_j1.insert(l1 + state.s[1]);
77  }
78  for (float j1 : array_j1) {
79 
80  for (auto &q : array_q) {
81  // q = final.m - initial.m
82 
83  float m0 = state.m[0] + q[0];
84  if (std::abs(m0) > j0) {
85  continue;
86  }
87 
88  float m1 = state.m[1] + q[1];
89  if (std::abs(m1) > j1) {
90  continue;
91  }
92 
93  StateTwo state_virtual = StateTwo(
94  state.species, {{n0, n1}}, {{l0, l1}}, {{j0, j1}}, {{m0, m1}});
95 
96  double energydiff = state.getEnergy() - state_virtual.getEnergy();
97  if (bfield != 0) {
98  energydiff += -bfield *
99  (cache.getMagneticDipole(state.first(), state.first()) +
100  cache.getMagneticDipole(state.second(), state.second()) -
101  cache.getMagneticDipole(state_virtual.first(),
102  state_virtual.first()) -
103  cache.getMagneticDipole(state_virtual.second(),
104  state_virtual.second()));
105  }
106 
107  C6 += std::pow(coulombs_constant *
108  array_angle_term[3 * (q[0] + 1) + (q[1] + 1)] *
109  cache.getElectricDipole(state_virtual.first(),
110  state.first()) *
111  cache.getElectricDipole(state_virtual.second(),
112  state.second()),
113  2) /
114  energydiff; // getMultipole(final, inital, 1)
115  }
116  }
117  }
118  }
119  }
120  }
121  }
122 
123  return C6;
124 }

126 eigen_dense_double_t PerturbativeInteraction::getC6(std::vector<StateTwo> states, double deltaN) {
127  eigen_dense_double_t C6_matrix = eigen_dense_double_t::Zero(states.size(), states.size());
128 
129  std::unordered_set<StateTwo> set_states(states.begin(), states.end());
130 
131  for (size_t idx_row = 0; idx_row < states.size(); ++idx_row) {
132  auto &state_row = states[idx_row];
133 
134  for (size_t idx_col = idx_row; idx_col < states.size(); ++idx_col) {
135  auto &state_col = states[idx_col];
136 
137  if (state_row.s[0] != state_col.s[0] || state_row.s[1] != state_col.s[1]) {
138  continue;
139  }
140  auto &s = state_col.s;
141 
142  if (state_row.species[0] != state_col.species[0] ||
143  state_row.species[1] != state_col.species[1]) {
144  continue;
145  }
146  auto &species = state_col.species;
147 
148  int n0_min = std::min(state_row.n[0], state_col.n[0]);
149  int n0_max = std::max(state_row.n[0], state_col.n[0]);
150  int n1_min = std::min(state_row.n[1], state_col.n[1]);
151  int n1_max = std::max(state_row.n[1], state_col.n[1]);
152 
153  double C6 = 0;
154 
155  for (int n0 = n0_min - deltaN; n0 <= n0_max + deltaN; ++n0) {
156 
157  for (int n1 = n1_min - deltaN; n1 <= n1_max + deltaN; ++n1) {
158 
159  std::vector<int> array_l0;
160  if (state_row.l[0] == state_col.l[0]) {
161  if (state_col.l[0] < n0 - 1) {
162  array_l0.push_back(state_col.l[0] + 1);
163  }
164  if (state_col.l[0] > 0) {
165  array_l0.push_back(state_col.l[0] - 1);
166  }
167  } else if (state_row.l[0] + 2 == state_col.l[0] && state_col.l[0] < n0 - 1) {
168  array_l0.push_back(state_col.l[0] + 1);
169  } else if (state_row.l[0] - 2 == state_col.l[0] && state_col.l[0] > 0) {
170  array_l0.push_back(state_col.l[0] - 1);
171  }
172  for (int l0 : array_l0) {
173 
174  std::vector<int> array_l1;
175  if (state_row.l[1] == state_col.l[1]) {
176  if (state_col.l[1] < n1 - 1) {
177  array_l1.push_back(state_col.l[1] + 1);
178  }
179  if (state_col.l[1] > 0) {
180  array_l1.push_back(state_col.l[1] - 1);
181  }
182  } else if (state_row.l[1] + 2 == state_col.l[1] &&
183  state_col.l[1] < n1 - 1) {
184  array_l1.push_back(state_col.l[1] + 1);
185  } else if (state_row.l[1] - 2 == state_col.l[1] && state_col.l[1] > 0) {
186  array_l1.push_back(state_col.l[1] - 1);
187  }
188  for (int l1 : array_l1) {
189 
190  std::set<float> array_j0;
191  if (std::abs(std::abs(l0 - s[0]) - state_col.j[0]) < 2 &&
192  std::abs(std::abs(l0 - s[0]) - state_row.j[0]) < 2) {
193  array_j0.insert(std::abs(l0 - s[0]));
194  }
195  if (std::abs(l0 + s[0] - state_col.j[0]) < 2 &&
196  std::abs(l0 + s[0] - state_row.j[0]) < 2) {
197  array_j0.insert(l0 + s[0]);
198  }
199  for (float j0 : array_j0) {
200 
201  std::set<float> array_j1;
202  if (std::abs(std::abs(l1 - s[1]) - state_col.j[1]) < 2 &&
203  std::abs(std::abs(l1 - s[1]) - state_row.j[1]) < 2) {
204  array_j1.insert(std::abs(l1 - s[1]));
205  }
206  if (std::abs(l1 + s[1] - state_col.j[1]) < 2 &&
207  std::abs(l1 + s[1] - state_row.j[1]) < 2) {
208  array_j1.insert(l1 + s[1]);
209  }
210  for (float j1 : array_j1) {
211 
212  for (auto &idx : array_q) {
213  int q0_forth = idx[0]; // q = final.m - initial.m
214  int q1_forth = idx[1];
215 
216  float m0 = state_col.m[0] + q0_forth;
217  if (std::abs(m0) > j0) {
218  continue;
219  }
220 
221  float m1 = state_col.m[1] + q1_forth;
222  if (std::abs(m1) > j1) {
223  continue;
224  }
225 
226  int q0_back = state_row.m[0] - m0;
227  int q1_back = state_row.m[1] - m1;
228  if (std::abs(q0_back) > 1 || std::abs(q1_back) > 1) {
229  continue;
230  }
231  if (array_q.size() == 3 && q0_back + q1_back != 0) {
232  continue;
233  }
234 
235  StateTwo state_virtual =
236  StateTwo(species, {{n0, n1}}, {{l0, l1}}, {{j0, j1}},
237  {{m0, m1}});
238  if (set_states.find(state_virtual) != set_states.end()) {
239  continue;
240  }
241 
242  double energydiff_row =
243  state_row.getEnergy() - state_virtual.getEnergy();
244  if (bfield != 0) {
245  energydiff_row += -bfield *
246  (cache.getMagneticDipole(state_row.first(),
247  state_row.first()) +
248  cache.getMagneticDipole(state_row.second(),
249  state_row.second()) -
250  cache.getMagneticDipole(state_virtual.first(),
251  state_virtual.first()) -
252  cache.getMagneticDipole(state_virtual.second(),
253  state_virtual.second()));
254  }
255 
256  double energydiff_col =
257  state_col.getEnergy() - state_virtual.getEnergy();
258  if (bfield != 0) {
259  energydiff_col += -bfield *
260  (cache.getMagneticDipole(state_col.first(),
261  state_col.first()) +
262  cache.getMagneticDipole(state_col.second(),
263  state_col.second()) -
264  cache.getMagneticDipole(state_virtual.first(),
265  state_virtual.first()) -
266  cache.getMagneticDipole(state_virtual.second(),
267  state_virtual.second()));
268  }
269 
270  C6 += coulombs_constant *
271  array_angle_term[3 * (q0_back + 1) + (q1_back + 1)] *
272  cache.getElectricDipole(state_row.first(),
273  state_virtual.first()) *
274  cache.getElectricDipole(state_row.second(),
275  state_virtual.second()) *
276  coulombs_constant *
277  array_angle_term[3 * (q0_forth + 1) + (q1_forth + 1)] *
278  cache.getElectricDipole(state_virtual.first(),
279  state_col.first()) *
280  cache.getElectricDipole(state_virtual.second(),
281  state_col.second()) *
282  0.5 *
283  (1 / energydiff_row +
284  1 / energydiff_col); // getMultipole(final, inital, 1)
285  }
286  }
287  }
288  }
289  }
290  }
291  }
292 
293  C6_matrix(idx_row, idx_col) = C6;
294  }
295  }
296 
297  return C6_matrix.selfadjointView<Eigen::Upper>();
298 }

300 eigen_dense_double_t PerturbativeInteraction::getC3(std::vector<StateTwo> states) {
301  eigen_dense_double_t C3_matrix = eigen_dense_double_t::Zero(states.size(), states.size());
302 
303  for (size_t idx_row = 0; idx_row < states.size(); ++idx_row) {
304  auto &state_row = states[idx_row];
305 
306  for (size_t idx_col = idx_row + 1; idx_col < states.size(); ++idx_col) {
307  auto &state_col = states[idx_col];
308 
309  int q0 = state_row.m[0] - state_col.m[0];
310  int q1 = state_row.m[1] - state_col.m[1];
311 
312  if (array_q.size() == 3 && q0 + q1 != 0) {
313  continue;
314  }
315  if (std::abs(q0) > 1 || std::abs(q1) > 1) {
316  continue;
317  }
318 
319  C3_matrix(idx_row, idx_col) = coulombs_constant *
320  array_angle_term[3 * (q0 + 1) + (q1 + 1)] *
321  cache.getElectricDipole(state_row.first(), state_col.first()) *
322  cache.getElectricDipole(state_row.second(), state_col.second());
323  }
324  }
325 
326  return C3_matrix.selfadjointView<Eigen::Upper>();
327 }

329 eigen_dense_double_t PerturbativeInteraction::getEnergies(std::vector<StateTwo> states) {
330  eigen_dense_double_t energies_matrix = eigen_dense_double_t::Zero(states.size(), states.size());
331 
332  for (size_t idx = 0; idx < states.size(); ++idx) {
333  auto &state = states[idx];
334 
335  double energy = state.getEnergy();
336  if (bfield != 0) {
337  energy += -bfield *
338  (cache.getMagneticDipole(state.first(), state.first()) +
339  cache.getMagneticDipole(state.second(), state.second()));
340  }
341  energies_matrix(idx, idx) = energy;
342  }
343 
344  return energies_matrix;

347 void PerturbativeInteraction::initializeAngleTerms(double angle) {
348  if (angle == 0) {
349  array_q.reserve(3);
350 
351  array_q.push_back({{0, 0}});
352  array_angle_term[3 * (0 + 1) + (0 + 1)] = -2;
353 
354  array_q.push_back({{1, -1}});
355  array_angle_term[3 * (1 + 1) + (-1 + 1)] = -1;
356 
357  array_q.push_back({{-1, 1}});
358  array_angle_term[3 * (-1 + 1) + (1 + 1)] = -1;
359  } else {
360  array_q.reserve(9);
361 
362  array_q.push_back({{0, 0}});
363  array_angle_term[3 * (0 + 1) + (0 + 1)] = 1. - 3. * std::pow(std::cos(angle), 2);
364 
365  array_q.push_back({{1, -1}});
366  array_angle_term[3 * (1 + 1) + (-1 + 1)] = -1 + 1.5 * std::pow(std::sin(angle), 2);
367 
368  array_q.push_back({{-1, 1}});
369  array_angle_term[3 * (-1 + 1) + (1 + 1)] = -1 + 1.5 * std::pow(std::sin(angle), 2);
370 
371  array_q.push_back({{1, 1}});
372  array_angle_term[3 * (1 + 1) + (1 + 1)] = -1.5 * std::pow(std::sin(angle), 2);
373 
374  array_q.push_back({{-1, -1}});
375  array_angle_term[3 * (-1 + 1) + (-1 + 1)] = -1.5 * std::pow(std::sin(angle), 2);
376 
377  array_q.push_back({{0, 1}});
378  array_angle_term[3 * (0 + 1) + (1 + 1)] =
379  3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
380 
381  array_q.push_back({{1, 0}});
382  array_angle_term[3 * (1 + 1) + (0 + 1)] =
383  3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
384 
385  array_q.push_back({{0, -1}});
386  array_angle_term[3 * (0 + 1) + (-1 + 1)] =
387  -3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
388 
389  array_q.push_back({{-1, 0}});
390  array_angle_term[3 * (-1 + 1) + (0 + 1)] =
391  -3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
392  }
